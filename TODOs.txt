# a "while" loop can be nested under an "if"
# a "if/elif/else" can also be nested under a "while" loop
print("on a 1-10 scale, how well are you today? (type and press [enter])")
happiness = int(input())
if happiness > 5:
    i = 0
    while i < happiness:
        print("happy!")
        i = i + 1
